def md5encode(url):
    if url.endswith("/"):
        path = "eps/api/resourceOperations/uploadsecretKeyIbuilding"

    encodetext = url + path

    input_name.update(encodetext.encode("utf-8"))
    return (input_name.hexdigest()).upper()
